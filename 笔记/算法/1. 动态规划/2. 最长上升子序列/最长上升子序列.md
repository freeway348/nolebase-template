### 例题

#### 最长上升子序列
![](assets/Pasted%20image%2020250506211924.png)

代码：
```C++
#include <iostream>
#include <algorithm>

using namespace std;

const int N = 5010;
int n;
int f[N]; // f[i]：以第 i 个数结尾的序列中，最长上升子序列的长度
int a[N];

int main()
{
    cin >> n;
    for (int i = 1; i <= n; i ++)
        cin >> a[i];
    for (int i = 1; i <= n; i ++)
    {
        f[i] = 1; // 先讨论 a[i] 之前没有比 a[i] 更大的数的情况
        for (int j = 1; j < i; j ++) // 再从头讨论倒数第二个不相同的值为 f[j] 的情况
            if (a[i] > a[j])
                f[i] = max(f[i], f[j] + 1);
    }

    int res = 0;
    for (int i = 1; i <= n; i ++)
        res = max(res, f[i]);
    cout << res << endl;
    return 0;
}
```
